		YAxis yAxis = new YAxis(new LinearNumericalAxisScaling(0.0, 35.0, yInterval1, yInterval2, yInterval3), "Y Axis");

		XYChart lineChart = new XYChart(values, "Gradiant", yAxis, xAxis);

		lineChart.setSize(1000, 500);
